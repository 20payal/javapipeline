pipeline {
  agent any

  stages {
    
      stage('Code checkout') {
      steps {
        
        git 'https://github.com/20payal/javapipeline.git'
      }

    }
    stage('Build') {
      steps {
        
        bat 'javac sample_code'
      }

    }
    stage('Deploy') {
      steps {
        
        bat 'java sample_code'

      }
    }
  }
}
